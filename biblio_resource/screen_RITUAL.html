1.1 Evento rituale/occasione

Datatype: InformationService

CA element code: ritual_event_ME

Ripetibilità: Sì

Quicktip: Campo Linked Data: indicare, se pertinente, uno o più eventi rituali o occasioni relativi alla risorsa etnomusicologica. Cercare inserendo almeno tre lettere per attivare la ricerca, ad esempio 'car' per 'Carnevale di Ivrea'; 'Now' o 'Naw' (o qualsiasi grafia alternativa con cui il nome è registrato su wikidata, es. central kurdish 'نه ورۆز') per 'Nowruz'.

Campo Linked Data: Per i termini ammissibili, aggiornati al , clicca sul seguente link per eseguire la query in tempo reale: Esegui Query (attenzione: per query complesse occorre attendere diversi secondi prima della restituzione dei risultati).

SELECT DISTINCT
  ?item
  ?keyLabel
  ?keyDesc
  ?lang
  ?layer
  ?placeLabel
WHERE {
  # Gate di curatela (solo focus)
  ?item wdt:P5008 wd:Q120325256 .

  # Match IT/EN su label o altLabel
  {
    ?item (rdfs:label|skos:altLabel) ?lab .
    FILTER(LANG(?lab) = "it")
    FILTER(REGEX(STR(?lab), "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)", "i"))
    BIND("it" AS ?lang)
    BIND(?lab AS ?rawKeyLabel)
  }
  UNION
  {
    ?item (rdfs:label|skos:altLabel) ?lab .
    FILTER(LANG(?lab) = "en")
    FILTER(REGEX(STR(?lab), "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)", "i"))
    BIND("en" AS ?lang)
    BIND(?lab AS ?rawKeyLabel)
  }

  # Description (fallback incrociato IT-EN)
  OPTIONAL { ?item schema:description ?descIt FILTER(LANG(?descIt)="it") }
  OPTIONAL { ?item schema:description ?descEn FILTER(LANG(?descEn)="en") }
  BIND(
    IF(?lang="it", COALESCE(?descIt, ?descEn),
                 COALESCE(?descEn, ?descIt)
    ) AS ?rawKeyDesc
  )

  # Luogo opzionale (location, se c'è)
  OPTIONAL {
    ?item wdt:P276 ?place .
    OPTIONAL { ?place rdfs:label ?pIt FILTER(LANG(?pIt)="it") }
    OPTIONAL { ?place rdfs:label ?pEn FILTER(LANG(?pEn)="en") }
    BIND(STR(COALESCE(?pIt, ?pEn, "")) AS ?placeLabel)
  }

  # Layer fisso
  BIND("ritual_occasion" AS ?layer)

  # Stringhe plain per CA
  BIND(STR(?rawKeyLabel) AS ?keyLabel)
  BIND(STR(COALESCE(?rawKeyDesc, "")) AS ?keyDesc)
}
ORDER BY ?lang ?keyLabel
LIMIT 200

1.2 Termine relativo al Personaggio/Ruolo rituale

Datatype: InformationService

CA element code: ritual_role_ME

Ripetibilità: Sì

Quicktip: Campo Linked Data: indicare, se pertinente, uno o più ruoli o personaggi parte di un rituale relativi alla risorsa etnomusicologica. Cercare inserendo almeno tre lettere per attivare la ricerca, ad esempio 'gen' per 'Generale del Carnevale di Ivrea'; 'mas' (o qualsiasi grafia alternativa con cui il nome è registrato su wikidata) per 'mascher (carnevale di Bagolino)' o altri risultati con le medesime iniziali.

Campo Linked Data: Per i termini ammissibili, aggiornati al , clicca sul seguente link per eseguire la query in tempo reale: Esegui Query (attenzione: per query complesse occorre attendere diversi secondi prima della restituzione dei risultati).

SELECT DISTINCT
  ?item
  ?keyLabel
  ?keyDesc
  ?lang
  ?layer
  ?layerOrder
  ?context ?contextLabel
WHERE {

  # 0) Vocabolario controllato (curatela)
  ?item wdt:P5008 wd:Q120325256 .
  ?item wdt:P2579 wd:Q208365 .

  # 1) Match su label/altLabel (IT/EN) con regex iniziale
  {
    ?item (rdfs:label|skos:altLabel) ?lab .
    FILTER(LANG(?lab) = "it")
    FILTER(REGEX(STR(?lab), "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)", "i"))
    BIND("it" AS ?lang)
    BIND(?lab AS ?rawKeyLabel)
  }
  UNION
  {
    ?item (rdfs:label|skos:altLabel) ?lab .
    FILTER(LANG(?lab) = "en")
    FILTER(REGEX(STR(?lab), "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)", "i"))
    BIND("en" AS ?lang)
    BIND(?lab AS ?rawKeyLabel)
  }

  # 2) Description coerente con la lingua del ramo (fallback incrociato)
  OPTIONAL { ?item schema:description ?descIt FILTER(LANG(?descIt) = "it") }
  OPTIONAL { ?item schema:description ?descEn FILTER(LANG(?descEn) = "en") }

  BIND(
    IF(?lang="it", COALESCE(?descIt, ?descEn),
                 COALESCE(?descEn, ?descIt)
    ) AS ?rawKeyDesc
  )

  # 3) Contesto opzionale: "part of" (solo per mostrare, NON per filtrare)
  OPTIONAL {
    ?item wdt:P361 ?context .
    OPTIONAL { ?context rdfs:label ?cIt FILTER(LANG(?cIt)="it") }
    OPTIONAL { ?context rdfs:label ?cEn FILTER(LANG(?cEn)="en") }
    BIND(STR(COALESCE(?cIt, ?cEn, STRAFTER(STR(?context), "/entity/"))) AS ?contextLabel)
  }

  # 4) Layer (euristico ma leggibile; puoi aggiungere classi man mano)
  OPTIONAL {
    ?item wdt:P31/wdt:P279* wd:Q97466613 .   # carnival character
    BIND(1 AS ?oCarnivalChar)
  }
  OPTIONAL {
    ?item wdt:P31/wdt:P279* wd:Q4752840 .    # fife and drum corps
    BIND(2 AS ?oFifeDrums)
  }

  BIND(
    IF(BOUND(?oCarnivalChar), 1,
      IF(BOUND(?oFifeDrums), 2,
        9
      )
    ) AS ?layerOrder
  )

  BIND(
    IF(?layerOrder = 1, "ritual_role",
      IF(?layerOrder = 2, "ritual_music",
        "ritual_term"
      )
    ) AS ?layer
  )

  # 5) Stringhe "plain" per CA
  BIND(STR(?rawKeyLabel) AS ?keyLabel)
  BIND(STR(COALESCE(?rawKeyDesc, "")) AS ?keyDesc)
}
ORDER BY ?layerOrder ?lang ?keyLabel
LIMIT 200

1.3 Termine relativo all'atto rituale

Datatype: InformationService

CA element code: ritual_act_ME

Ripetibilità: Sì

Quicktip: Campo Linked Data: indicare, se pertinente, uno o più termini relativi all'atto rituale relativi alla risorsa etnomusicologica. Cercare inserendo almeno tre lettere per attivare la ricerca, ad esempio 'inc' per 's'Incravamentu (Bosa) o 's'Incravamentu (Santu Lussurgiu); 'abb' per 'abbruciamento degli scarli (Carnevale di Ivrea).

Campo Linked Data: Per i termini ammissibili, aggiornati al , clicca sul seguente link per eseguire la query in tempo reale: Esegui Query (attenzione: per query complesse occorre attendere diversi secondi prima della restituzione dei risultati).

SELECT DISTINCT ?item ?itemLabel ?itemDescription ?lang
WHERE {
    ?item wdt:P5008 wd:Q120325256.
    ?item wdt:P2579 wd:Q208365.
    ?item(rdfs:label|skos:altLabel) ?itemLabel.
    FILTER(REGEX(?itemLabel, "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)", "i"))
    FILTER(LANGMATCHES(LANG(?itemLabel), "it") || LANGMATCHES(LANG(?itemLabel), "en"))
    ?item schema:description ?itemDescription.
    FILTER(LANGMATCHES(LANG(?itemDescription), "it") || LANGMATCHES(LANG(?itemDescription), "en"))
    FILTER(LANG(?itemLabel) = LANG(?itemDescription))
   
  BIND(LANG(?itemLabel) AS ?lang)
  SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE], en, it". }  

} 
order by ?item
