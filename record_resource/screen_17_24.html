SELECT DISTINCT ?item ?itemLabel ?qvalue ?itemDescription ?value ?url ?lang
WHERE {
  
   ?item wdt:P31 wd:Q47451145.
    ?item wdt:P528 ?folktale_type.
    ?item wdt:P2540 ?value;
          (rdfs:label|skos:altLabel) ?itemLabel.
    FILTER(REGEX(?itemLabel, "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)", "i"))
FILTER(LANGMATCHES(LANG(?itemLabel), "it") || LANGMATCHES(LANG(?itemLabel), "en"))
    ?item schema:description ?itemDescription.
    FILTER(LANGMATCHES(LANG(?itemDescription), "it") || LANGMATCHES(LANG(?itemDescription), "en"))
    FILTER(LANG(?itemLabel) = LANG(?itemDescription))  
    
    OPTIONAL{ ?item wdt:P31 wd:Q47451145.
    ?item wdt:P528 ?folktale_type.
    ?item wdt:P2540 ?value;
    p:P2540/pq:P1810 ?qvalue.
            }
    OPTIONAL{?item wdt:P31 wd:Q47451145.
    ?item wdt:P528 ?folktale_type.
    ?item wdt:P2540 ?value;
    p:P2540/pq:P4970 ?qvalue.
            }
    OPTIONAL {?item wdt:P31 wd:Q47451145.
    ?item wdt:P528 ?folktale_type.
    ?item wdt:P2540 ?value;
    p:P2540/pq:P1810 ?qvalue.
    FILTER (REGEX(?qvalue, "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)", "i"))
    BIND(IF(CONTAINS(?itemLabel, "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)"), COALESCE(?itemLabel, ?qvalue), ?qvalue) AS ?displayLabel)
   
    }
    OPTIONAL {
    ?item wdt:P31 wd:Q47451145.
    ?item wdt:P528 ?folktale_type.
    ?item wdt:P2540 ?value;
    p:P2540/pq:P4970 ?qvalue.
    FILTER (REGEX(?qvalue, "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)", "i"))
    BIND(IF(CONTAINS(?itemLabel, "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)"), COALESCE(?itemLabel, ?qvalue), ?qvalue) AS ?displayLabel)}
    
    OPTIONAL {?item wdt:P31 wd:Q47451145.
    ?item wdt:P528 ?folktale_type.
    ?item wdt:P2540 ?value;
    p:P2540/pq:P1810 ?qvalue.
    FILTER (REGEX(?value, "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)", "i"))}
 OPTIONAL {
    ?item wdt:P31 wd:Q47451145.
    ?item wdt:P528 ?folktale_type.

                }

  CONCAT(
    "ATU Index: ", 
    IF(BOUND(?qvalue), ?qvalue, ""),

   CONCAT("http://www.mftd.org/index.php?action=atu&src=atu&id=", ?value)

}
