1.2 Materiali

Datatype: InformationService

CA element code: materials_ME

Ripetibilità: Sì

Quicktip: Campo Linked Data. Si indicano qui i materiali di cui è costituito lo strumento, attraverso un collegamento alla terminologia disponibile su Wikidata. Per un elenco dei termini disponsibili consultare la documentazione.

Campo Linked Data: Per i termini ammissibili, aggiornati al , clicca sul seguente link per eseguire la query in tempo reale: Esegui Query (attenzione: per query complesse occorre attendere diversi secondi prima della restituzione dei risultati).

SELECT DISTINCT ?item ?itemLabel ?itemDescription ?value ?url ?lang
WHERE {
  {
    ?item ?p ?value .
    VALUES ?p {wdt:P4801 wdt:P9112}
    ?item (rdfs:label|skos:altLabel) ?itemLabel.
    FILTER (REGEX(?value, "mmaterial|3dmat|graphicspsm", "i"))
    FILTER(REGEX(?itemLabel, "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)", "i"))
    FILTER(LANGMATCHES(LANG(?itemLabel), "it"))
    OPTIONAL {
      ?item schema:description ?itemDescription.
      FILTER(LANGMATCHES(LANG(?itemDescription), "it"))
    }
    BIND("it" AS ?lang)
  }
  UNION
  {
    ?item ?p ?value .
    VALUES ?p {wdt:P4801 wdt:P9112 wdt:P9236}
    ?item (rdfs:label|skos:altLabel) ?itemLabel.
    FILTER (REGEX(?value, "mmaterial|3dmat|graphicsspsm", "i"))
    FILTER(REGEX(?itemLabel, "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)", "i"))
    FILTER(LANGMATCHES(LANG(?itemLabel), "en"))
    OPTIONAL {
      ?item schema:description ?itemDescription.
     FILTER(LANGMATCHES(LANG(?itemDescription), "en"))
    }
    BIND("en" AS ?lang)
  }
  UNION
  { 
    #?item ?p ?value .
    ?item wdt:P1014 ?value .
    #VALUES ?p {wdt:P4801 wdt:P9112 wdt:P1014}
    VALUES ?item {wd:Q226697 wd:Q11472 wd:Q1143474}
    VALUES ?value {"300014161" "300014109" "300011851"}
    ?item (rdfs:label|skos:altLabel) ?itemLabel.
    #FILTER (REGEX(?value, "mmaterial|3dmat|graphicspsm", "i"))
    FILTER(REGEX(?itemLabel, "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)", "i"))
    FILTER(LANGMATCHES(LANG(?itemLabel), "it"))
    OPTIONAL {
      ?item schema:description ?itemDescription.
      FILTER(LANGMATCHES(LANG(?itemDescription), "it"))
    }
    BIND("it" AS ?lang)
  }
UNION
  { 
    #?item ?p ?value .
    ?item wdt:P1014 ?value .
    #VALUES ?p {wdt:P4801 wdt:P9112 wdt:P1014}
    VALUES ?item {wd:Q226697 wd:Q11472 wd:Q1143474}
    VALUES ?value {"300014161" "300014109" "300011851"}
    ?item (rdfs:label|skos:altLabel) ?itemLabel.
    #FILTER (REGEX(?value, "mmaterial|3dmat|graphicspsm", "i"))
    FILTER(REGEX(?itemLabel, "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)", "i"))
    FILTER(LANGMATCHES(LANG(?itemLabel), "en"))
    OPTIONAL {
      ?item schema:description ?itemDescription.
      FILTER(LANGMATCHES(LANG(?itemDescription), "en"))
    }
    BIND("it" AS ?lang)
  }
  SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE], en, it". }
 BIND(
    IF(REGEX(?value, "mmaterial", "i"), 
      CONCAT("https://id.loc.gov/vocabulary/", ?value),
      IF(REGEX(?value, "3dmat|graphicspsm", "i"),
        CONCAT("https://www.iflastandards.info/unimarc/terms/", ?value),
        "-"
      )
    ) AS ?url
  ).
} 
order by ?item
