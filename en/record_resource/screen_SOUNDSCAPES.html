SELECT DISTINCT ?item ?itemLabel ?itemDescription ?envo ?lang
WHERE {
  {
    ?item wdt:P3859 ?envo ;
          rdfs:label ?itemLabel .
    FILTER(LANG(?itemLabel) = "it")
    FILTER(REGEX(STR(?itemLabel), "(^|\\W)(***PLACEHOLDER***)(\\W|$)", "i"))
    BIND("it" AS ?lang)
  }
  UNION
  {
    ?item wdt:P3859 ?envo ;
          rdfs:label ?itemLabel .
    FILTER(LANG(?itemLabel) = "en")
    FILTER(REGEX(STR(?itemLabel), "(^|\\W)(***PLACEHOLDER***)(\\W|$)", "i"))
    BIND("en" AS ?lang)
  }
  UNION
  {
    # Se vuoi che funzioni anche digitando direttamente un codice ENVO (es. ENVO:00000446)
    ?item wdt:P3859 ?envo .
    FILTER(REGEX(STR(?envo), "***PLACEHOLDER***", "i"))
  }

  OPTIONAL {
    ?item schema:description ?itemDescription .
    FILTER(LANG(?itemDescription) IN ("it","en"))
  }

  SERVICE wikibase:label { bd:serviceParam wikibase:language "it, en". }
}
ORDER BY ?item

1.3 SPL and temporal localization

Datatype: Container

CA element code: quantitativedata_CN

Quicktip: Container section devoted to collecting quantitative data on the soundscape. Includes: minimum, average and maximum sound pressure levels (SPLs), expressed in decibels (dB); date and time of observation or recording, useful for contextualizing the soundscape over time. These data can be collected with phonometric instruments, digital recording devices or noted in a structured manner during field campaigns. Their combination with qualitative perceptions allows for the integration of ecoacoustic, bioacoustic and ethnographic approaches in the description of soundscapes.

1.3.4 Date and time

Datatype: DateRange

CA element code: soundscape_timing_ME

Quicktip: daterange' field, enter single dates or, for uncertain dates, time intervals. Examples: 'August 30, 1746', '1900', 'June 7, 2007 at 15:31:48', 'June 2016'. For uncertain dates, use the following formats: 'about 1955', '1955 ~ 3y' (1955 plus or minus 3 years), 'about 1950 - 1956', 'June 10, 1955 ~ 10d' (June 10, 1955 plus or minus 10 days), '850 BCE', '40 CE', '6/10/????' or '06/10/?' (date with day and month indicated, but no year), '6/????' (date with month indicated, but no year). The following values can also be used, which will then be transformed by the system into a range of actual dates: 'autumn 2003' (season indication), '20 Q3' (third quarter of the twentieth century).

Datatype: InformationService

SELECT DISTINCT ?item ?itemLabel ?itemDescription ?value ?scientificName ?lang ?num
WHERE {
  {
    SELECT DISTINCT ?item ?lang ?num WHERE {
      {
        SERVICE wikibase:mwapi {
          bd:serviceParam wikibase:endpoint "www.wikidata.org";
                          wikibase:api "EntitySearch";
                          mwapi:search "***PLACEHOLDER***";
                          mwapi:language "it";
                          mwapi:limit "20".
          ?item wikibase:apiOutputItem mwapi:item.
          ?num  wikibase:apiOrdinal true.
        }
        BIND("it" AS ?lang)
      }
      UNION
      {
        SERVICE wikibase:mwapi {
          bd:serviceParam wikibase:endpoint "www.wikidata.org";
                          wikibase:api "EntitySearch";
                          mwapi:search "***PLACEHOLDER***";
                          mwapi:language "en";
                          mwapi:limit "20".
          ?item wikibase:apiOutputItem mwapi:item.
          ?num  wikibase:apiOrdinal true.
        }
        BIND("en" AS ?lang)
      }
    }
    ORDER BY ?lang ?num
    LIMIT 40
  }

  # deve avere Catalogue of Life ID (P10585)
  ?item wdt:P10585 ?value .

  OPTIONAL { ?item wdt:P225 ?scientificName . }

  OPTIONAL {
    ?item schema:description ?itemDescription .
    FILTER(LANG(?itemDescription) IN ("it","en"))
  }

  SERVICE wikibase:label { bd:serviceParam wikibase:language "it,en". }
}
ORDER BY ?lang ?num

1.5.2 Sound

Datatype: InformationService

CA element code: sound_rel_term_ME

Quicktip: nan

Linked Data field: for the allowed terms, updated on , click the following link to run the query in real time: Run Query (for complex queries, wait several seconds before results are returned).

SELECT DISTINCT
  ?item
  ?label
  ?description
  ?caLabel
  ?caDescription
  ?keyLabel
  ?keyDesc
  ?itemLabel
  ?itemDescription
  ?lang ?num
  ?layer ?layerOrder
  ?producedSound ?producedSoundLabel
WHERE {
  {
    SELECT DISTINCT ?item ?lang ?num WHERE {
      {
        SERVICE wikibase:mwapi {
          bd:serviceParam wikibase:endpoint "www.wikidata.org";
                          wikibase:api "EntitySearch";
                          mwapi:search "***PLACEHOLDER***";
                          mwapi:language "it";
                          mwapi:limit "40".
          ?item wikibase:apiOutputItem mwapi:item.
          ?num  wikibase:apiOrdinal true.
        }
        BIND("it" AS ?lang)
      }
      UNION
      {
        SERVICE wikibase:mwapi {
          bd:serviceParam wikibase:endpoint "www.wikidata.org";
                          wikibase:api "EntitySearch";
                          mwapi:search "***PLACEHOLDER***";
                          mwapi:language "en";
                          mwapi:limit "100".
          ?item wikibase:apiOutputItem mwapi:item.
          ?num  wikibase:apiOrdinal true.
        }
        BIND("en" AS ?lang)
      }
      UNION
      {
        SERVICE wikibase:mwapi {
          bd:serviceParam wikibase:endpoint "www.wikidata.org";
                          wikibase:api "EntitySearch";
                          mwapi:search "***PLACEHOLDER***";
                          mwapi:language "se";
                          mwapi:limit "40".
          ?item wikibase:apiOutputItem mwapi:item.
          ?num  wikibase:apiOrdinal true.
        }
        BIND("se" AS ?lang)
      }
    }
    ORDER BY ?lang ?num
    LIMIT 180
  }

  # Labels per lingua (per key coerente col ramo)
  OPTIONAL { ?item rdfs:label ?labelIt FILTER(LANG(?labelIt) = "it") }
  OPTIONAL { ?item rdfs:label ?labelEn FILTER(LANG(?labelEn) = "en") }
  OPTIONAL { ?item rdfs:label ?labelSe FILTER(LANG(?labelSe) = "se") }

  # Query string in lowercase
  BIND(LCASE("***PLACEHOLDER***") AS ?q)

  # Tieni SOLO il ramo dove la label nella lingua del ramo contiene davvero la query
  FILTER(
    ( ?lang = "it" && BOUND(?labelIt) && CONTAINS(LCASE(STR(?labelIt)), ?q) ) ||
    ( ?lang = "en" && BOUND(?labelEn) && CONTAINS(LCASE(STR(?labelEn)), ?q) ) ||
    ( ?lang = "se" && BOUND(?labelSe) && CONTAINS(LCASE(STR(?labelSe)), ?q) )
  )

  # Descrizioni per lingua (fallback: se -> en)
  OPTIONAL { ?item schema:description ?descIt FILTER(LANG(?descIt) = "it") }
  OPTIONAL { ?item schema:description ?descEn FILTER(LANG(?descEn) = "en") }
  OPTIONAL { ?item schema:description ?descSe FILTER(LANG(?descSe) = "se") }

  BIND(
    IF(?lang="it", COALESCE(?descIt, ?descEn),
      IF(?lang="en", COALESCE(?descEn, ?descIt),
                    COALESCE(?descSe, ?descEn, ?descIt)
      )
    ) AS ?tmpDesc
  )

  # layer classification
  OPTIONAL {
    VALUES ?soundRoot { wd:Q11461 wd:Q179448 }
    ?item (wdt:P279*|wdt:P31/wdt:P279*) ?soundRoot .
    BIND(1 AS ?oSound)
  }
  OPTIONAL {
    ?item (wdt:P279*|wdt:P31/wdt:P279*) wd:Q16332653 .
    BIND(2 AS ?oMeteo)
  }
  OPTIONAL {
    VALUES ?anthroRoot { wd:Q4323994 wd:Q360418 wd:Q42889 wd:Q11019 }
    ?item (wdt:P279*|wdt:P31/wdt:P279*) ?anthroRoot .
    BIND(3 AS ?oAnthro)
  }
  OPTIONAL {
    ?item wdt:P4733 ?producedSound .
    BIND(3 AS ?oAnthro2)
  }
  OPTIONAL {
    ?item (wdt:P279*|wdt:P31/wdt:P279*) wd:Q16521 .
    BIND(4 AS ?oTaxon)
  }
  OPTIONAL {
    VALUES ?eventRoot { wd:Q13418847 wd:Q1656682 }  # historical event / event
    ?item (wdt:P279*|wdt:P31/wdt:P279*) ?eventRoot .
    BIND(5 AS ?oEvent)
  }

  BIND(
    IF(BOUND(?oSound), 1,
      IF(BOUND(?oMeteo), 2,
        IF(BOUND(?oAnthro) || BOUND(?oAnthro2), 3,
          IF(BOUND(?oTaxon), 4,
            IF(BOUND(?oEvent), 5, 99)
          )
        )
      )
    ) AS ?layerOrder
  )
  FILTER(?layerOrder < 99)

  BIND(
    IF(?layerOrder = 1, "sound_type",
      IF(?layerOrder = 2, "source_meteo",
        IF(?layerOrder = 3, "source_anthro",
          IF(?layerOrder = 4, "source_taxon",
            "event_context"
          )
        )
      )
    ) AS ?layer
  )

  # Label service: fornisce ?itemLabel / ?itemDescription (+ producedSoundLabel)
  SERVICE wikibase:label { bd:serviceParam wikibase:language "it,en,se". }

  # keyLabel coerente col ramo + fallback su itemLabel + fallback su QID (stringa nuda)
  BIND(
    STR(
      COALESCE(
        IF(?lang="it", ?labelIt,
          IF(?lang="en", ?labelEn, ?labelSe)
        ),
        ?itemLabel,
        STRAFTER(STR(?item), "/entity/")
      )
    ) AS ?keyLabel
  )

  # keyDesc robusta: tmpDesc -> itemDescription -> "" (stringa nuda)
  BIND(STR(COALESCE(?tmpDesc, ?itemDescription, "")) AS ?keyDesc)

  # Alias “CA-friendly” SENZA toccare itemLabel/itemDescription
  BIND(?keyLabel AS ?label)
  BIND(?keyDesc  AS ?description)
  BIND(?keyLabel AS ?caLabel)
  BIND(?keyDesc  AS ?caDescription)
}
ORDER BY ?layerOrder ?lang ?num
LIMIT 100
