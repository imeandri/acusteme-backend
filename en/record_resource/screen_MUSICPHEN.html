1.1 Musical genre

Datatype: InformationService

CA element code: genre_ME

Repeatability: Yes

Quicktip: Linked Data field: indicate, if relevant, one or more musical genres related to the ethnomusicological resource. Search by entering at least three letters to activate the search.

Linked Data field: for the allowed terms, updated on , click the following link to run the query in real time: Run Query (for complex queries, wait several seconds before results are returned).

SELECT DISTINCT ?item ?itemLabel ?itemDescription ?lang
WHERE {
{
    ?item wdt:P5008 wd:Q120325256.
    ?item wdt:P2579 wd:Q208365.
    ?item(rdfs:label|skos:altLabel) ?itemLabel.
    FILTER(REGEX(?itemLabel, "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)", "i"))
}
  UNION
  {
    ?item ?p ?value.
    VALUES ?p {wdt:P9218 wdt:P9219}
    ?item(rdfs:label|skos:altLabel) ?itemLabel.
    FILTER(REGEX(?itemLabel, "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)", "i"))
    }
    ?item wdt:P31 wd:Q188451 .
    FILTER(LANGMATCHES(LANG(?itemLabel), "it") || LANGMATCHES(LANG(?itemLabel), "en"))
    ?item schema:description ?itemDescription.
    FILTER(LANGMATCHES(LANG(?itemDescription), "it") || LANGMATCHES(LANG(?itemDescription), "en"))
    FILTER(LANG(?itemLabel) = LANG(?itemDescription))
   
  BIND(LANG(?itemLabel) AS ?lang)
  SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE], en, it". } 
} 
order by ?item

1.3.5 Hornbostel-Sachs suffix (1)

Datatype: InformationService

CA element code: hornbostel_sachs_suffix

Quicktip: Linked Data field: indicate, if relevant, the HS classification of the musical instrument related to the ethnomusicological resource. Search by entering at least three letters to activate the search.

Linked Data field: for the allowed terms, updated on , click the following link to run the query in real time: Run Query (for complex queries, wait several seconds before results are returned).

SELECT DISTINCT
?qid ?url
?hs ?taxon ?classLabel
?search
#?desc
WHERE {
  {
    ?root wdt:P31 wd:Q134778286 .
    VALUES ?p {p:P1762 p:P13598}
   
    ?root ?p ?statement .
    ?statement ps:P1762|ps:P13598 ?hs .
    OPTIONAL { ?statement pq:P13044 ?class . }
   
    ?root wdt:P1448 ?taxon .
    FILTER (LANG(?taxon) = "it")

    OPTIONAL {
      ?class rdfs:label ?classLabel .
      FILTER (LANG(?classLabel) = "it")
    }

    BIND(CONCAT(
      ?hs, " ", ?taxon,
      IF(BOUND(?classLabel), CONCAT(" [", ?classLabel, "]"), "")
    ) AS ?search)
   
    FILTER (
    REGEX(?search, "***PLACEHOLDER***", "i")
  )
  }
  UNION
  {
    ?root wdt:P31 wd:Q134778286 .
    ?item wdt:P279* ?root .

    VALUES ?p {p:P1762 p:P13598}
    ?item ?p ?statement .
    ?statement ps:P1762|ps:P13598 ?hs .
    OPTIONAL { ?statement pq:P13044 ?class . }
   
    ?item wdt:P1448 ?taxon .
    FILTER (LANG(?taxon) = "it")

    OPTIONAL {
      ?class rdfs:label ?classLabel .
      FILTER (LANG(?classLabel) = "it")
    }

    BIND(CONCAT(
      ?hs, " ", ?taxon,
      IF(BOUND(?classLabel), CONCAT(" [", ?classLabel, "]"), "")
    ) AS ?search)
   
    FILTER (
    REGEX(?search, "***PLACEHOLDER***", "i")
  )
  }

  BIND(COALESCE(STRAFTER(STR(?item), "entity/"), STRAFTER(STR(?root), "entity/")) AS ?qid)
  BIND(IRI(CONCAT("http://www.wikidata.org/entity/", ?qid)) AS ?url)

  SERVICE wikibase:label {
    bd:serviceParam wikibase:language "[AUTO_LANGUAGE],it,en".
    #?root schema:description ?rootDescription.
    #?item schema:description ?itemDescription.
  }
  #BIND(COALESCE(?itemDescription, ?rootDescription) AS ?desc)
}
ORDER BY ?hs

  SERVICE wikibase:mwapi {
    bd:serviceParam wikibase:endpoint "www.wikidata.org";
                    wikibase:api "EntitySearch";
                    mwapi:search "***PLACEHOLDER***";
                    mwapi:language "it".
    ?item wikibase:apiOutputItem mwapi:item.
    ?num wikibase:apiOrdinal true.
  }

  OPTIONAL { ?item schema:description ?dEn FILTER(LANG(?dEn)="en") }
