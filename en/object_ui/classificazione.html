1.2 Hornbostel-Sachs classification

Datatype: InformationService

CA element code: hs_wikidata_ME

Repeatability: Yes

Quicktip: Linked Data field. Enter at least three letters to start the search for the appropriate taxon and choose from the alternatives proposed in the portions of the HS tree displayed and filtered for search words. The HS classification revised by Febo Guizzi, in Italian, is used. Further details can be found in the documentation.

Linked Data field: for the allowed terms, updated on , click the following link to run the query in real time: Run Query (for complex queries, wait several seconds before results are returned).

SELECT distinct ?item ?itemLabel ?hs ?search
WHERE {
 
  ?item wdt:P31 wd:Q1254773 ;
        wdt:P361 ?classificazione .
  VALUES ?classificazione {wd:Q53424891  wd:Q496327  wd:Q26836418}
 
  ?item ?p ?hs .
  VALUES ?p {wdt:P1762 wdt:P13598}
 
  ?item wdt:P1448 ?taxon .
  FILTER (LANG(?taxon) = "it")
 
  BIND(CONCAT(?hs, " ", ?taxon) AS ?search)
 
FILTER (
    REGEX(?hs, "***PLACEHOLDER***", "i") ||
    REGEX(?taxon, "***PLACEHOLDER***", "i") ||
    REGEX(?itemLabel, "***PLACEHOLDER***")
  )
  SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE], it". }  
 
}
ORDER BY ?hs

1.3 HS suffix

Datatype: InformationService

CA element code: hs_suffix_ME

Repeatability: Yes

Quicktip: Linked Data field. Enter at least three letters to start the search for the appropriate taxon and choose from the alternatives proposed in the portions of the HS tree displayed and filtered for search words. The HS classification revised by Febo Guizzi, in Italian, is used. Further details can be found in the documentation.

Linked Data field: for the allowed terms, updated on , click the following link to run the query in real time: Run Query (for complex queries, wait several seconds before results are returned).

SELECT DISTINCT
?qid ?url
?hs ?taxon ?classLabel
?search
#?desc
WHERE {
  {
    ?root wdt:P31 wd:Q134778286 .
    VALUES ?p {p:P1762 p:P13598}
   
    ?root ?p ?statement .
    ?statement ps:P1762|ps:P13598 ?hs .
    OPTIONAL { ?statement pq:P13044 ?class . }
   
    ?root wdt:P1448 ?taxon .
    FILTER (LANG(?taxon) = "it")

    OPTIONAL {
      ?class rdfs:label ?classLabel .
      FILTER (LANG(?classLabel) = "it")
    }

    BIND(CONCAT(
      ?hs, " ", ?taxon,
      IF(BOUND(?classLabel), CONCAT(" [", ?classLabel, "]"), "")
    ) AS ?search)
   
    FILTER (
    REGEX(?search, "***PLACEHOLDER***", "i")
  )
  }
  UNION
  {
    ?root wdt:P31 wd:Q134778286 .
    ?item wdt:P279* ?root .

    VALUES ?p {p:P1762 p:P13598}
    ?item ?p ?statement .
    ?statement ps:P1762|ps:P13598 ?hs .
    OPTIONAL { ?statement pq:P13044 ?class . }
   
    ?item wdt:P1448 ?taxon .
    FILTER (LANG(?taxon) = "it")

    OPTIONAL {
      ?class rdfs:label ?classLabel .
      FILTER (LANG(?classLabel) = "it")
    }

    BIND(CONCAT(
      ?hs, " ", ?taxon,
      IF(BOUND(?classLabel), CONCAT(" [", ?classLabel, "]"), "")
    ) AS ?search)
   
    FILTER (
    REGEX(?search, "***PLACEHOLDER***", "i")
  )
  }

  BIND(COALESCE(STRAFTER(STR(?item), "entity/"), STRAFTER(STR(?root), "entity/")) AS ?qid)
  BIND(IRI(CONCAT("http://www.wikidata.org/entity/", ?qid)) AS ?url)

  SERVICE wikibase:label {
    bd:serviceParam wikibase:language "[AUTO_LANGUAGE],it,en".
    #?root schema:description ?rootDescription.
    #?item schema:description ?itemDescription.
  }
  #BIND(COALESCE(?itemDescription, ?rootDescription) AS ?desc)
}
ORDER BY ?hs

1.5.2 Typological definition (LOD)

Datatype: InformationService

CA element code: typologicaldef_lod_ME

Quicktip: Linked Data field. Enter at least three letters and choose from the proposed alternatives. Refer to the documentation for a complete list of available type definitions and for requests to enter new type definitions.

Linked Data field: for the allowed terms, updated on , click the following link to run the query in real time: Run Query (for complex queries, wait several seconds before results are returned).

SELECT DISTINCT ?item ?itemLabel ?itemDescription ?lang
WHERE {
    ?item wdt:P1762 ?value .
    ?item(rdfs:label|skos:altLabel) ?itemLabel .
    FILTER(REGEX(?itemLabel, "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)", "i"))
    
    FILTER(LANGMATCHES(LANG(?itemLabel), "it") || LANGMATCHES(LANG(?itemLabel), "en"))
    ?item schema:description ?itemDescription.
    FILTER(LANGMATCHES(LANG(?itemDescription), "it") || LANGMATCHES(LANG(?itemDescription), "en"))
    FILTER(LANG(?itemLabel) = LANG(?itemDescription))
   
  BIND(LANG(?itemLabel) AS ?lang)
  SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE], en, it". } 
} 
order by ?item

1.10 Medium of performance (LoC)

Datatype: InformationService

CA element code: INST_mediumofperf_ME

Quicktip: Linked Data field. Allows entry, by initiating a search with at least three letters, of the name of a musical instrument retrieved from Library of Congress vocabularies, linking ethno-organological classification to biblliographic area vocabularies useful, for example, during bilbiographical enrichment on the navigation portal. Used primarily for Eurocult area instruments.

Linked Data field: for the allowed terms, updated on , click the following link to run the query in real time: Run Query (for complex queries, wait several seconds before results are returned).

SELECT DISTINCT ?item ?itemLabel ?itemDescription ?value ?lang
WHERE {
    ?item wdt:P8516 ?value.
    #FILTER (REGEX(?value, "mp", "i"))
        ?item (rdfs:label|skos:altLabel) ?itemLabel.
    FILTER(REGEX(?itemLabel, "^(***PLACEHOLDER***|[^ ]+ ***PLACEHOLDER***)", "i"))
    ?item(rdfs:label|skos:altLabel) ?itemLabel.
     
    FILTER(LANGMATCHES(LANG(?itemLabel), "it") || LANGMATCHES(LANG(?itemLabel), "en"))
    ?item schema:description ?itemDescription.
    FILTER(LANGMATCHES(LANG(?itemDescription), "it") || LANGMATCHES(LANG(?itemDescription), "en"))
    FILTER(LANG(?itemLabel) = LANG(?itemDescription))
   
  BIND(LANG(?itemLabel) AS ?lang)
  SERVICE wikibase:label { bd:serviceParam wikibase:language "[AUTO_LANGUAGE], en, it". } 

} 
order by ?item
